#  ARIMA Modeling of Disposable Personal Income (PDI)
# Author: Missael's AI Companion
# Goal: Fit ARIMA model to PDI data with full diagnostics and forecasting

#  STEP 1: Import necessary libraries
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from statsmodels.tsa.stattools import adfuller
from statsmodels.graphics.tsaplots import plot_acf, plot_pacf
from statsmodels.tsa.arima.model import ARIMA
from statsmodels.stats.diagnostic import acorr_ljungbox
from scipy import stats
import warnings
warnings.filterwarnings("ignore")
sns.set(style="whitegrid")

#  STEP 2: Upload Excel file
from google.colab import files
uploaded = files.upload()

#  STEP 3: Load and preprocess the data
# Assumes Excel file has columns: YEAR, GDP, PDI, PCE, PROFITS, DIVIDENDS
df = pd.read_excel(next(iter(uploaded)))

# Convert YEAR to quarterly PeriodIndex
df['YEAR'] = df['YEAR'].astype(str).str.replace('Q', 'Q')  # Keep 'YYYYQn' format
df['YEAR'] = pd.PeriodIndex(df['YEAR'], freq='Q')          # Convert to quarterly PeriodIndex
df.set_index('YEAR', inplace=True)

#  STEP 4: Plot the original PDI time series
plt.figure(figsize=(10, 4))
plt.plot(df.index.to_timestamp(), df['PDI'], label='Disposable Personal Income (PDI)', color='blue')
plt.title('PDI Time Series')
plt.xlabel('Time')
plt.ylabel('PDI')
plt.legend()
plt.show()

#  STEP 5: ADF Test for stationarity
result = adfuller(df['PDI'])
print("ADF Statistic:", result[0])
print("p-value:", result[1])
for key, value in result[4].items():
    print(f"Critical Value ({key}): {value}")
if result[1] > 0.05:
    print(" PDI is non-stationary. Differencing required.")
else:
    print(" PDI is stationary.")

#  STEP 6: Differencing if needed
d = 0
pdi_series = df['PDI']
while adfuller(pdi_series.dropna())[1] > 0.05:
    pdi_series = pdi_series.diff().dropna()
    d += 1
print(f"Order of differencing required: d = {d}")

#  STEP 7: Plot ACF and PACF to identify AR and MA orders
fig, ax = plt.subplots(1, 2, figsize=(12, 4))
plot_acf(pdi_series, ax=ax[0], lags=20)
plot_pacf(pdi_series, ax=ax[1], lags=20)
ax[0].set_title('ACF of Differenced Series')
ax[1].set_title('PACF of Differenced Series')
plt.tight_layout()
plt.show()

#  STEP 8: Fit several ARIMA models and compare AIC/BIC
order_candidates = [(0,d,0),(1,d,1), (2,d,1), (1,d,2), (2,d,2), (3,d,1), (1,d,3)]
results = []
for order in order_candidates:
    model = ARIMA(df['PDI'], order=order)
    fitted = model.fit()
    results.append((order, fitted.aic, fitted.bic))
    print(f"ARIMA{order}  AIC: {fitted.aic:.2f}, BIC: {fitted.bic:.2f}")

#  STEP 9: Choose best model based on AIC/BIC
best_model = min(results, key=lambda x: x[1])
print(f"\n Best model based on AIC: ARIMA{best_model[0]}")

#  STEP 10: Fit final model and show summary
final_model = ARIMA(df['PDI'], order=best_model[0]).fit()
print(final_model.summary())

#  STEP 11: Residual diagnostics
residuals = final_model.resid

# Plot residuals
fig, ax = plt.subplots(1, 2, figsize=(12, 4))
sns.histplot(residuals, kde=True, ax=ax[0], color='purple')
stats.probplot(residuals, dist="norm", plot=ax[1])
ax[0].set_title("Residual Histogram")
ax[1].set_title("Q-Q Plot of Residuals")
plt.tight_layout()
plt.show()

# Ljung-Box test for autocorrelation
ljung = acorr_ljungbox(residuals, lags=[10], return_df=True)
print("\nLjung-Box Test (lag=10):")
print(ljung)

#  STEP 12: Forecasting
forecast = final_model.get_forecast(steps=8)
forecast_mean = forecast.predicted_mean
conf_int = forecast.conf_int()

# Plot forecast
plt.figure(figsize=(10, 4))
plt.plot(df.index.to_timestamp(), df['PDI'], label='Historical PDI')
plt.plot(forecast_mean.index.to_timestamp(), forecast_mean, label='Forecast', color='green')
plt.fill_between(forecast_mean.index.to_timestamp(), conf_int.iloc[:, 0], conf_int.iloc[:, 1], color='lightgreen', alpha=0.5)
plt.title('PDI Forecast with 95% Confidence Interval')
plt.xlabel('Time')
plt.ylabel('PDI')
plt.legend()
plt.show()

#  STEP 13: Interpretation
print("\n Interpretation of Final ARIMA Model:")
print(f"Model Order: ARIMA{best_model[0]}")
print("AR Coefficients:", final_model.arparams)
print("MA Coefficients:", final_model.maparams)
print("P-values of coefficients:")
print(final_model.pvalues)
print("\nResiduals appear approximately normal and uncorrelated (see plots and Ljung-Box test).")
print("Model is suitable for short-term forecasting of PDI.")
